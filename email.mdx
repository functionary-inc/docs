<Note>

Sends a email to the address specified with the content and subject specified.

<ParamField body="type" type="string = 'email'" required>
  Must equal the string `email`.
</ParamField>

<ParamField body="args" type="object" required>
  <Expandable title="properties" defaultOpen={true}>

    <ResponseField name="email" type="templatable string" required>
      email address
    </ResponseField>

    <ResponseField name="text" type="templatable string" required>
      the text of the body of the email message.
    </ResponseField>

    <ResponseField name="subject" type="templatable string" required>
      the subject of the email message.
    </ResponseField>

  </Expandable>
</ParamField>

```json
{
  "type": "email",
  "args": {
    "email": "{{modelProperties.email}}",
    "subject": "Top 10 Tips to Finish Integration!"
    "text": "Hey {{displayName}} -- Hope all is well. - Justin"
  }
}
```
