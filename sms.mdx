<Note>

Sends a text to the number specified with the content specified.

<ParamField body="type" type="string = 'sms'" required>
  Must equal the string `sms`.
</ParamField>

<ParamField body="args" type="object" required>
  <Expandable title="properties" defaultOpen={true}>

    <ResponseField name="number" type="templatable string" required>
      number to send the text, including the country code (ie +13876658364). No dashes or seperations.
    </ResponseField>

    <ResponseField name="content" type="templatable string" required>
      the content of the text message.
    </ResponseField>

  </Expandable>
  
</ParamField>

```json
{
  "type": "sms",
  "args": {
    "number": "{{modelProperties.number}}",
    "content": "Congrats {{displayName}}!"
  }
}
```
